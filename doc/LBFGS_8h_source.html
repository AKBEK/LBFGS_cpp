78  template <typename Foo>
79  inline int minimize(Foo& f, Vector& x, Scalar& fx)
80  {
81  const int n = x.size();
82  const int fpast = m_param.past;
83  reset(n);
84 
85  // Evaluate function and compute gradient
86  fx = f(x, m_grad);
87  Scalar xnorm = x.norm();
88  Scalar gnorm = m_grad.norm();
89  if(fpast > 0)
90  m_fx[0] = fx;
91 
92  // Early exit if the initial x is already a minimizer
93  if(gnorm <= m_param.epsilon * std::max(xnorm, Scalar(1.0)))
94  {
95  return 1;
96  }
97 
98  // Initial direction
99  m_drt.noalias() = -m_grad;
100  // Initial step
101  Scalar step = Scalar(1.0) / m_drt.norm();
102 
103  int k = 1;
104  int end = 0;
105  for( ; ; )
106  {
107  // Save the curent x and gradient
108  m_xp.noalias() = x;
109  m_gradp.noalias() = m_grad;
110 
111  // Line search to update x, fx and gradient
112  LineSearch<Scalar>::Backtracking(f, fx, x, m_grad, step, m_drt, m_xp, m_param);
113 
114  // New x norm and gradient norm
115  xnorm = x.norm();
116  gnorm = m_grad.norm();
117 
118  // Convergence test -- gradient
119  if(gnorm <= m_param.epsilon * std::max(xnorm, Scalar(1.0)))
120  {
121  return k;
122  }
123  // Convergence test -- objective function value
124  if(fpast > 0)
125  {
126  if(k >= fpast && (m_fx[k % fpast] - fx) / fx < m_param.delta)
127  return k;
128 
129  m_fx[k % fpast] = fx;
130  }
131  // Maximum number of iterations
132  if(m_param.max_iterations != 0 && k >= m_param.max_iterations)
133  {
134  return k;
135  }
136 
137  // Update s and y
138  // s_{k+1} = x_{k+1} - x_k
139  // y_{k+1} = g_{k+1} - g_k
140  MapVec svec(&m_s(0, end), n);
141  MapVec yvec(&m_y(0, end), n);
142  svec.noalias() = x - m_xp;
143  yvec.noalias() = m_grad - m_gradp;
144 
145  // ys = y's = 1/rho
146  // yy = y'y
147  Scalar ys = yvec.dot(svec);
148  Scalar yy = yvec.squaredNorm();
149  m_ys[end] = ys;
150 
151  // Recursive formula to compute d = -H * g
152  m_drt.noalias() = -m_grad;
153  int bound = std::min(m_param.m, k);
154  end = (end + 1) % m_param.m;
155  int j = end;
156  for(int i = 0; i < bound; i++)
157  {
158  j = (j + m_param.m - 1) % m_param.m;
159  MapVec sj(&m_s(0, j), n);
160  MapVec yj(&m_y(0, j), n);
161  m_alpha[j] = sj.dot(m_drt) / m_ys[j];
162  m_drt.noalias() -= m_alpha[j] * yj;
163  }
164 
165  m_drt *= (ys / yy);
166 
167  for(int i = 0; i < bound; i++)
168  {
169  MapVec sj(&m_s(0, j), n);
170  MapVec yj(&m_y(0, j), n);
171  Scalar beta = yj.dot(m_drt) / m_ys[j];
172  m_drt.noalias() += (m_alpha[j] - beta) * sj;
173  j = (j + 1) % m_param.m;
174  }
175 
176  // step = 1.0 as initial guess
177  step = Scalar(1.0);
178  k++;
179  }
180 
181  return k;
182  }
